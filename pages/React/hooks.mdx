// Example
import { useHuddle01 } from "@huddle01/react"

const App = () => {
	const { initialize, isInitialized, onEvent } = useHuddle01();

	useEffect(() => {
		if(!isInitialized)
			initialize("YOUR_PROJECT_ID");
	}. []);

	onEvent("lobby:joined", () => {
		console.log("Lobby Joined Successfully");
	})

	return (
		<div>
			Hello World!
		</div>
	)
}
